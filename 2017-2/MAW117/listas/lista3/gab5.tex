	(x^2-4)y & = & x\\
	yx^2-4y-x & = & 0 \\
	x(xy-1)-4y & = & 0\\
	x^2y-x-4y & = & 0,
\end{eqnarray*}

	f^{-1}(y)=\left\{\begin{array}{l}

			0,\mbox{ se }y=0.\end{array}\right.
\]
